# Mod_PageSpeed-Auto-Installer
Automatically Install Mod_PageSpeed in CentOS Web Panel

wget -c http://dl-package.bullten.in/cwp/cwp-modpagespeed.sh\

chmod +x cwp-modpagespeed.sh

printf 'Ny' | sh cwp-modpagespeed.sh
